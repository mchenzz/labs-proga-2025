Задание 1 — Факториал числа:

#include <iostream>

int main() {
    int n;
    unsigned long long fact = 1;
    std::cout << "Введите число: ";
    std::cin >> n;
    for (int i = 1; i <= n; ++i) {
        fact *= i;
    }
    std::cout << "Факториал: " << fact << std::endl;
    return 0;
}

Задание 3 — Проверка на простое число:

#include <iostream>

int main() {
    int n, i;
    bool isPrime = true;
    std::cout << "Введите число: ";
    std::cin >> n;
    if (n <= 1) isPrime = false;
    for (i = 2; i * i <= n; ++i) {
        if (n % i == 0) {
            isPrime = false;
            break;
        }
    }
    if (isPrime)
        std::cout << "Число простое" << std::endl;
    else
        std::cout << "Число не является простым" << std::endl;
    return 0;
}


Задание 4 — Сумма цифр числа:

#include <iostream>
#include <cstdlib>

int main() {
    int n, sum = 0;
    std::cout << "Введите число: ";
    std::cin >> n;
    n = abs(n);
    while (n > 0) {
        sum += n % 10;
        n /= 10;
    }
    std::cout << "Сумма цифр: " << sum << std::endl;
    return 0;
}


Задание 5 — Перевод в двоичную систему:

#include <iostream>
#include <vector>

int main() {
    int n;
    std::cout << "Введите число: ";
    std::cin >> n;
    if (n == 0) {
        std::cout << "0" << std::endl;
        return 0;
    }
    std::vector<int> bin;
    unsigned int x = n;
    while (x > 0) {
        bin.push_back(x % 2);
        x /= 2;
    }
    std::cout << "В двоичной системе: ";
    for (int i = bin.size() - 1; i >= 0; --i) {
        std::cout << bin[i];
    }
    std::cout << std::endl;
    return 0;
}

Задание 7 — Ряд Фибоначчи:

#include <iostream>

int main() {
    int n;
    std::cout << "Введите N: ";
    std::cin >> n;
    unsigned long long a = 0, b = 1;
    std::cout << "Ряд Фибоначчи: ";
    for (int i = 0; i < n; ++i) {
        std::cout << a << " ";
        unsigned long long next = a + b;
        a = b;
        b = next;
    }
    std::cout << std::endl;
    return 0;
}


Задание 8 — Подсчет гласных в строке:

#include <iostream>
#include <string>
#include <cctype>

int main() {
    std::string str;
    std::cout << "Введите строку: ";
    std::getline(std::cin, str);
    int count = 0;
    for (char c : str) {
        c = std::tolower(c);
        if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') count++;
    }
    std::cout << "Количество гласных: " << count << std::endl;
    return 0;
}


Задание 9 — Реверс числа:

#include <iostream>

int main() {
    int n, rev = 0;
    std::cout << "Введите число: ";
    std::cin >> n;
    int orig = n;
    n = abs(n);
    while (n > 0) {
        rev = rev * 10 + n % 10;
        n /= 10;
    }
    if (orig < 0) rev = -rev;
    std::cout << "В обратном порядке: " << rev << std::endl;
    return 0;
}


Задание 10 — Калькулятор с повторением:

#include <iostream>
#include <string>

int main() {
    while (true) {
        std::string command;
        double a, b, result;
        char op;
        std::cout << "Введите операцию в формате: число операция число (или напишите 'выход' для выхода): ";
        std::cin >> command;
        if (command == "выход") break;
        try {
            a = std::stod(command);
        } catch (...) {
            std::cout << "Ошибка ввода" << std::endl;
            continue;
        }
        std::cin >> op >> b;

        if (op == '+')
            result = a + b;
        else if (op == '-')
            result = a - b;
        else if (op == '*')
            result = a * b;
        else if (op == '/') {
            if (b == 0) {
                std::cout << "Деление на ноль!" << std::endl;
                continue;
            }
            result = a / b;
        } else {
            std::cout << "Неизвестная операция!" << std::endl;
            continue;
        }

        std::cout << "Результат: " << result << std::endl;
    }
    std::cout << "Калькулятор завершен." << std::endl;
    return 0;
}
